# Alert

## Usage

```vue
<script setup>
  import { useNotifier } from 'vuetify-notifier'

  const notifier = useNotifier()

  function onShow(){
    notifier.alert({
      title: 'Warning',
      text: 'This is alert content',
      color: 'error',
      onSubmit(){
        // Do something when button click
      }
    })
  }
</script>

<template>
  <!-- ... -->
    <VBtn @click="onShow">Show</VBtn>
  <!-- ... -->
</template>
```
